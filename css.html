CSS Cheatsheet

/* CSS Cheatsheet */
/* Universal Selector */
/* Resets margin and padding to 0 and includes box-sizing: border-box globally */
* {
margin: 0;
padding: 0;
box-sizing: border-box;
}
/* Type Selector */
/* Styles all h1 elements with a larger font size */
h1 {
font-size: 2em;
}
/* Class Selector */
/* Styles elements with the class "className" with a specific text color */
.className {
color: #333;
}
/* ID Selector */
/* Styles the element with the ID "uniqueId" with a specific background color */
#uniqueId {
background-color: #f0f0f0;
}
/* Descendant Selector */
/* Styles child elements inside parentElement with a border */
parentElement childElement {
border: 1px solid #ccc;
}
/* Pseudo-class Selector */
/* Adds an underline to links on hover */
a:hover {
text-decoration: underline;
}
/* Grouping Selector */
/* Applies bold font weight to h2, h3, and h4 elements */
h2, h3, h4 {
font-weight: bold;
}
/* Text Styling */
/* Defines various text styling properties */
color: #333;
font-size: 16px;
font-family: 'Arial', sans-serif;
font-weight: normal;
line-height: 1.5;
text-align: center;
text-decoration: none;
text-transform: uppercase;
/* Box Model */
/* Defines properties related to the box model: width, height, margin, padding, border */
width: 100px;
height: 100px;
margin: 10px;
padding: 20px;
border: 1px solid #ccc;
box-sizing: border-box;
/* Display */
/* Defines how elements are displayed in the layout */
display: block;
display: inline;
display: inline-block;
display: flex;
display: grid;
display: none;
/* Positioning */
/* Defines the position and stacking order of elements */
position: static;
position: relative;
position: absolute;
position: fixed;
top: 10px;
right: 20px;
bottom: 30px;
left: 40px;
z-index: 1;
/* Background */
/* Defines background properties, including color, image, and positioning */
background-color: #f0f0f0;
background-image: url('image.jpg');
background-size: cover;
background-position: center;
background-repeat: no-repeat;
/* Flexbox */
/* Defines properties for flexible box layout */
display: flex;
flex-direction: row;
justify-content: space-between;
align-items: center;
/* Grid */
/* Defines properties for grid layout */
display: grid;
grid-template-columns: repeat(3, 1fr);
grid-gap: 10px;
/* Transitions and Animations */
/* Defines properties for smooth transitions and animations */
transition: property duration timing-function;
animation: name duration timing-function iteration-count direction;
/* Responsive Design */
/* Media query for smaller screens */
@media (max-width: 768px) {
/* CSS rules for smaller screens */
} /* Comments */
/* This is a comment */
/* Importing External Stylesheets */
/* Imports an external stylesheet named 'external-stylesheet.css' */
@import url('external-stylesheet.css');
